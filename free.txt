1. Création des topics Kafka
kafka-topics.sh --create --zookeeper localhost:2181 \
--replication-factor 1 --partitions 1 \
--topic topic_clients

kafka-topics.sh --create --zookeeper localhost:2181 \
--replication-factor 1 --partitions 1 \
--topic topic_produits

kafka-topics.sh --create --zookeeper localhost:2181 \
--replication-factor 1 --partitions 1 \
--topic topic_commandes

kafka-topics.sh --create --zookeeper localhost:2181 \
--replication-factor 1 --partitions 1 \
--topic topic_paiements

2. Lister les topics
kafka-topics.sh --list --zookeeper localhost:2181
3. Producer
cat /root/data/clients.json | kafka-console-producer.sh \
--broker-list localhost:9092 \
--topic topic_clients

cat /root/data/produits.json | kafka-console-producer.sh \
--broker-list localhost:9092 \
--topic topic_produits

cat /root/data/commandes.json | kafka-console-producer.sh \
--broker-list localhost:9092 \
--topic topic_commandes

cat /root/data/paiements.json | kafka-console-producer.sh \
--broker-list localhost:9092 \
--topic topic_paiements

4. Consumer
kafka-console-consumer.sh --zookeeper localhost:2181 \
--topic topic_clients \
--from-beginning

5. Consumer + fichier local
kafka-console-consumer.sh --zookeeper localhost:2181 \
--topic topic_clients \
--from-beginning \
--timeout-ms 10000 > /root/clients_kafka.json

Même logique pour les autres topics.

Requêtes Spark SQL 

Afficher tous les clients
spark.sql("SELECT * FROM clients").show()

Afficher les clients de sexe féminin
spark.sql("""
SELECT *
FROM clients
WHERE sexe = 'F'
""").show()

Afficher les clients âgés de plus de 23 ans
spark.sql("""
SELECT nom, prenom, age
FROM clients
WHERE age > 23
ORDER BY age DESC
""").show()

Afficher les produits de la catégorie Informatique
spark.sql("""
SELECT *
FROM produits
WHERE categorie = 'Informatique'
""").show()

Afficher le nom complet du client, le produit et la quantité
spark.sql("""
SELECT
c.nom,
c.prenom,
p.nom_produit,
co.quantite
FROM clients c
JOIN commandes co ON c.id_client = co.id_client
JOIN produits p ON co.id_produit = p.id_produit
""").show()

Calculer le montant total payé par chaque client
spark.sql("""
SELECT
c.nom,
c.prenom,
SUM(pa.montant) AS total_paye
FROM clients c
JOIN commandes co ON c.id_client = co.id_client
JOIN paiements pa ON co.id_commande = pa.id_commande
WHERE pa.statut = 'Payé'
GROUP BY c.nom, c.prenom
""").show()

Afficher le produit ayant le plus de commandes
spark.sql("""
SELECT
p.nom_produit,
COUNT(*) AS nb_commandes
FROM commandes co
JOIN produits p ON co.id_produit = p.id_produit
GROUP BY p.nom_produit
ORDER BY nb_commandes DESC
LIMIT 1
""").show()

Calculer le chiffre d’affaires moyen par produit
spark.sql("""
SELECT
p.nom_produit,
AVG(pa.montant) AS moyenne
FROM paiements pa
JOIN commandes co ON pa.id_commande = co.id_commande
JOIN produits p ON co.id_produit = p.id_produit
GROUP BY p.nom_produit
""").show()

Calculer le nombre de clients par sexe
spark.sql("""
SELECT
sexe,
COUNT(*) AS total
FROM clients
GROUP BY sexe
""").show()

Afficher les villes dont les recettes dépassent 50000 DH
spark.sql("""
SELECT
c.ville,
SUM(pa.montant) AS recettes
FROM clients c
JOIN commandes co ON c.id_client = co.id_client
JOIN paiements pa ON co.id_commande = pa.id_commande
WHERE pa.statut = 'Payé'
GROUP BY c.ville
HAVING SUM(pa.montant) > 50000
""").show()
